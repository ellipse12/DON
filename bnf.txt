<DON> ::= <expr>*
<expr> ::= <assign> | <directive> 
<assign> ::= <id> "=" <literal>
<directive> ::= "#" <id> <expr>
<id> ::= [*] 
<literal> ::= <number> | <string> | <bool> | "null" | "[" (<literal>* "," | <literal>) ]" | "{" <expr> "}"
<number> ::= <int> | <float> | <hex> | <octal> | <binary>
<int> ::= "-"? [1-9] [0-9]*
<float> ::= <int> "." [0-9]* | "." [0-9]* | <int> "."
<hex> ::= "0x" ([0-9] | [a-z] | [A-Z])*
<octal> ::= "0o" [0-7]*
<binary> ::= "0b" ("0" | "1")*
<bool> ::= "true" | "false"
<string> ::= "\"" [*] "\""
